255 // show image on external display
256 int ExternalDisplay::showImage(bb::ImageData imageData) {
257  int returnCode;
258  int i, j;
259  int dx, dy, dw, dh;
260  const unsigned char *imagePixels;
261  unsigned char *ptr = NULL;
262  int usage, stride;
263  int rect[4] = { 0, 0, 0, 0 };
264  int source_rect[4] = { 0, 0, 0, 0 };
265  screen_pixmap_t screen_pix;
266  screen_buffer_t screen_pbuf;
267 
268  // check if imageData is an existing object
269  if (!imageData.isValid()) {
270  perror("showImage");
271  return EXIT_FAILURE;
272  }
273 
274  // copy image data to screen buffer
275  imagePixels = imageData.constPixels();
276 
277  // center image on screen, i.e. upper left corner at (dx,dy) with original width dw and height dh
278  dx = (_width - imageData.width()) / 2;
279  dw = imageData.width();
280  if (dx < 0) {
281  dx = 0;
282  dw = _width;
283  }
284 
285  dy = (_height - imageData.height()) / 2;
286  dh = imageData.height();
287  if (dy < 0) {
288  dy = 0;
289  dh = _height;
290  }
291 
292  // define BLIT command to copy image data (source) to display (destination)
293  int hg[] = {
294  SCREEN_BLIT_SOURCE_WIDTH, imageData.width(),
295  SCREEN_BLIT_SOURCE_HEIGHT, imageData.height(),
296  SCREEN_BLIT_DESTINATION_X, dx,
297  SCREEN_BLIT_DESTINATION_Y, dy,
298  SCREEN_BLIT_DESTINATION_WIDTH, dw,
299  SCREEN_BLIT_DESTINATION_HEIGHT, dh,
300  SCREEN_BLIT_TRANSPARENCY, SCREEN_TRANSPARENCY_NONE,
301  SCREEN_BLIT_END
302  };
303 
304  rect[2] = _width;
305  rect[3] = _height;
306  source_rect[2] = imageData.width();
307  source_rect[3] = imageData.height();
308 
309  // create pixmap without actual buffer
310  returnCode = screen_create_pixmap(&screen_pix, _screen_ctx2);
311  if (returnCode != EXIT_SUCCESS) {
312  perror("screen_create_pixmap");
313  return EXIT_FAILURE;
314  }
315 
316  // set pixel format (here: RGBA = RGB pixels with alpha blending for transparency)
317  returnCode = screen_set_pixmap_property_iv(screen_pix, SCREEN_PROPERTY_FORMAT, &_pixelFormat);
318  if (returnCode != EXIT_SUCCESS) {
319  perror("screen_set_pixmap_property_iv: SCREEN_PROPERTY_FORMAT");
320  return EXIT_FAILURE;
321  }
322 
323  // set pixel buffer access method
324  usage = SCREEN_USAGE_WRITE | SCREEN_USAGE_NATIVE;
325  returnCode = screen_set_pixmap_property_iv(screen_pix, SCREEN_PROPERTY_USAGE, &usage);
326  if (returnCode != EXIT_SUCCESS) {
327  perror("screen_set_pixmap_property_iv: SCREEN_PROPERTY_USAGE");
328  return EXIT_FAILURE;
329  }
330 
331  // set size of pixel map according to image size
332  returnCode = screen_set_pixmap_property_iv(screen_pix, SCREEN_PROPERTY_BUFFER_SIZE, source_rect+2);
333  if (returnCode != EXIT_SUCCESS) {
334  perror("screen_set_pixmap_property_iv: SCREEN_PROPERTY_BUFFER_SIZE");
335  return EXIT_FAILURE;
336  }
337 
338  // create actual pixel buffer according to image size
339  returnCode = screen_create_pixmap_buffer(screen_pix);
340  if (returnCode != EXIT_SUCCESS) {
341  perror("screen_create_pixmap_buffer");
342  return EXIT_FAILURE;
343  }
344 
345  // retrieve pointer to the beginning of the screen buffer in the pixel map
346  returnCode = screen_get_pixmap_property_pv(screen_pix, SCREEN_PROPERTY_RENDER_BUFFERS, (void **)&screen_pbuf);
347  if (returnCode != EXIT_SUCCESS) {
348  perror("screen_get_pixmap_property_iv: SCREEN_PROPERTY_RENDER_BUFFERS");
349  return EXIT_FAILURE;
350  }
351 
352  // get actual pointer to the beginning of the screen buffer
353  returnCode = screen_get_buffer_property_pv(screen_pbuf, SCREEN_PROPERTY_POINTER, (void **)&ptr);
354  if (returnCode != EXIT_SUCCESS) {
355  perror("screen_get_buffer_property_iv: SCREEN_PROPERTY_POINTER");
356  return EXIT_FAILURE;
357  }
358 
359  // retrieve the number of bytes between two subsequent lines in the image (stride)
360  returnCode = screen_get_buffer_property_iv(screen_pbuf, SCREEN_PROPERTY_STRIDE, &stride);
361  if (returnCode != EXIT_SUCCESS) {
362  perror("screen_get_buffer_property_iv: SCREEN_PROPERTY_STRIDE");
363  return EXIT_FAILURE;
364  }
365 
366  // copy ImageData content to screen pixel buffer, reversing byte order RGBA -> BGRA
367  for (i = 0; i < source_rect[3]; i++, ptr += stride) {
368  for (j = 0; j < source_rect[2]; j++) {
369  ptr[j*4] = imagePixels[4*i*source_rect[2]+4*j+2];
370  ptr[j*4+1] = imagePixels[4*i*source_rect[2]+4*j+1];
371  ptr[j*4+2] = imagePixels[4*i*source_rect[2]+4*j+0];
372  ptr[j*4+3] = imagePixels[4*i*source_rect[2]+4*j+3];
373  }
374  }
375 
376  // copy buffer to screen window and make it visible
377  screen_blit(_screen_ctx2, _screen_buf[0], screen_pbuf, hg);
378  screen_post_window(_screen_win2, _screen_buf[0], 1, rect, 0);
379 
380  // free allocated resources of pixmap
381  screen_destroy_pixmap(screen_pix);
382 
383  return EXIT_SUCCESS;
384 }
