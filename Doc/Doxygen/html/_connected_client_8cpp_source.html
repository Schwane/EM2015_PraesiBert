232  void ConnectedClient::handleCmdRead()
233  {
234  QByteArray inputData;
235 
236  while (1)
237  {
238  if (!m_next_block_size_cmd)
239  {
240  int x = 4;
241 
242  while (x > 0)
243  {
244  inputData = m_cmdSocket->read(x);
245  x -= inputData.size();
246  }
247 
248  for(int i = 0; i < 4; ++i)
249  {
250  m_next_block_size_cmd |= ((quint8) inputData.at(i)) << i * 8;
251  }
252  }
253 
254  if (m_cmdSocket->bytesAvailable() < m_next_block_size_cmd)
255  return;
256  else
257  {
258  inputData.clear();
259  inputData = m_cmdSocket->read(qMin(m_next_block_size_cmd, m_cmdSocket->bytesAvailable()));
260  m_next_block_size_cmd = 0;
261  m_bufferCmd.append(inputData);
262  qDebug() << "data: " << m_bufferCmd;
263  }
264 
265  if (m_next_block_size_cmd == 0)
266  {
267  qDebug() << "New command at client.\n" << m_bufferCmd;
268  emit newCmd(m_bufferCmd, m_clientID);
269  m_bufferCmd.clear();
270  }
271 
272  if (m_cmdSocket -> bytesAvailable() == 0)
273  break;
274  }
