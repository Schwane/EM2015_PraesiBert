26 MessageAuthenticator::hmacSha1(QByteArray key, QByteArray baseString)
27 {
28  //Note: taken from http://wiki.qt.io/HMAC-SHA1
29  int blockSize = 64; // HMAC-SHA-1 block size, defined in SHA-1 standard
30  if (key.length() > blockSize) { // if key is longer than block size (64), reduce key length with SHA-1 compression
31  key = QCryptographicHash::hash(key, QCryptographicHash::Sha1);
32  }
33 
34  QByteArray innerPadding(blockSize, char(0x36)); // initialize inner padding with char "6"
35  QByteArray outerPadding(blockSize, char(0x5c)); // initialize outer padding with char "quot;
36  // ascii characters 0x36 ("6") and 0x5c ("quot;) are selected because they have large
37  // Hamming distance (http://en.wikipedia.org/wiki/Hamming_distance)
38 
39  for (int i = 0; i < key.length(); i++) {
40  innerPadding[i] = innerPadding[i] ^ key.at(i); // XOR operation between every byte in key and innerpadding, of key length
41  outerPadding[i] = outerPadding[i] ^ key.at(i); // XOR operation between every byte in key and outerpadding, of key length
42  }
43 
44  // result = hash ( outerPadding CONCAT hash ( innerPadding CONCAT baseString ) ).toBase64
45  QByteArray total = outerPadding;
46  QByteArray part = innerPadding;
47  part.append(baseString);
48  total.append(QCryptographicHash::hash(part, QCryptographicHash::Sha1));
49  QByteArray hashed = QCryptographicHash::hash(total, QCryptographicHash::Sha1);
50  QString b64 = hashed.toBase64();
51  hashed.clear();
52  //return always base64 encoded hash
53  hashed.append(b64);
54  return hashed;
55 }
