99 Client::setSlide(QMap<QString, QVariant> parameters, QMap<QString, QString> parameter_types)
100 {
101  Message *resp = new Message(CMD_ACK_RESPONSE, id,"server");
102  //Check if all parameters are contained and send error if not.
103  if(!parameters.contains("slide"))
104  {
105  resp -> addParameter("status",QString("error"));
106  resp -> addParameter("message",QString("Parameter: slide number not defined"));
107  return resp;
108  }
109 
110  if(!parameter_types.contains("slide") || parameter_types.value("slide") != "integer")
111  {
112  resp -> addParameter("status",QString("error"));
113  resp -> addParameter("message",QString("Parameter: slide - WRONG TYPE"));
114  return resp;
115  }
116 
117  //Check boundaries of slide and either set slide or emit error.
118  int slide = parameters.value("slide").toInt();
119 
120  if (slide < 0)
121  {
122  resp -> addParameter("status",QString("error"));
123  resp -> addParameter("message",QString("Parameter: slide - smaller than 0"));
124  return resp;
125  }
126 
127  if (prs->getTotalSlides() <= slide)
128  {
129  resp -> addParameter("status",QString("error"));
130  resp -> addParameter("message",QString("Parameter: slide - bigger than total slides"));
131  emit noMoreSlides();
132  return resp;
133  }
134 
135  prs->setSlide(slide);
136  resp -> addParameter("status", QString("ok"));
137 
138  return resp;
139 }

163 Message*
164 Client::loginResponse(QMap<QString, QVariant> parameters, QMap<QString, QString> parameter_types)
165 {
166  Message *resp = new Message(CMD_ACK_RESPONSE,"client","server");
167  //Check if parameters are contained
168  if (parameters.contains("status") && parameters.contains("id"))
169  {
170  QString status = parameters.value("status").toString();
171  QString id = parameters.value("id").toString();
172  if (status == "ok") //if status is ok then login is done and client id is set to that from server
173  {
174  login_state = ACCEPTED;
175  this -> id = id;
176  }
177  else
178  {
179  login_state = REJECTED;
180  }
181  emit loginStateChanged(); //inform that state has changed
182  resp -> addParameter("status", QString("ok"));
183  }
184  else //error
185  {
186  resp -> addParameter("status",QString("error"));
187  resp -> addParameter("message",QString("Parameter: status - no status contained"));
188  }
189 
190  return resp;
191 }
