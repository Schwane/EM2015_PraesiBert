59  bool ServerSocket::beginListening(QString cmdPort_str, QString dataPort_str)
60  {
61  // Define variable for IP-Address
62  QString ipAddress_str;
63 
64  // Fetch available IP-Addresses
65  QList<QHostAddress> ipList = QNetworkInterface::allAddresses();
66 
67  // Loop through list of available IPs and choose the first IPv4 that is not localhost
68  for (int i = 0; i < ipList.size(); ++i) {
69  if (ipList.at(i) != QHostAddress::LocalHost && ipList.at(i).toIPv4Address())
70  {
71  ipAddress_str = ipList.at(i).toString();
72  break;
73  }
74  }
75 
76  // If no IP was found, use localhost
77  if (ipAddress_str.isEmpty())
78  ipAddress_str = QHostAddress(QHostAddress::LocalHost).toString();
79 
80  // Emit signal with new IP-Address of the server
81  emit newIP(ipAddress_str);
82 
83  // Start listening for incoming connections on both sockets
84  if(!m_cmdServer->listen(QHostAddress::Any, cmdPort_str.toInt(0, 10)))
85  {
86  // Close socket and return Error-State, if listening failed
87  qDebug() << "Could not start the Command-Server. Closing sockets.\n";
88  m_cmdServer->close();
89  return false;
90  }
91 
92  if(!m_dataServer->listen(QHostAddress::Any, dataPort_str.toInt(0, 10)))
93  {
94  // Close socket and return Error-State, if listening failed
95  qDebug() << "Could not start the Data-Server. Closing sockets.\n";
96  m_cmdServer->close();
97  return false;
98  }
99 
100  connect(m_cmdServer, SIGNAL(newConnection()), this, SLOT(handleNewConnection()));
101  connect(m_dataServer, SIGNAL(newConnection()), this, SLOT(handleNewConnection()));
102 
103  qDebug() << "Listening for incoming command connections on Port " << cmdPort_str << ".\n";
104  qDebug() << "Listening for incoming data connections on Port " << dataPort_str << ".\n";
105  qDebug() << "IP-Address is: " << ipAddress_str << ".\n";
106  return true;
107  }

287  void ServerSocket::handleNewConnection()
288  {
289  QTcpServer* signalSender;
290  signalSender = qobject_cast<QTcpServer*>(sender());
291 
292  QTcpSocket* newSocket;
293 
294  if(signalSender == m_cmdServer)
295  {
296  newSocket = m_cmdServer->nextPendingConnection();
297  }
298  if(signalSender == m_dataServer)
299  {
300  newSocket = m_dataServer->nextPendingConnection();
301  }
302 
303  /*
304  * Check, if a Client with the same peer address is already in the clientList.
305  * If so, just add the new socket type to the client.
306  * Otherwise a new ConnectedClient Object will be created later.
307  */
308  for(int i = 0; i < m_clientList.length(); i++)
309  {
310  if(m_clientList.at(i)->getPeerAddress() == newSocket->peerAddress())
311  {
312  qDebug() << "Found existing Client with ID " << m_clientList.at(i)->getClientID() << " and IP-Address " << m_clientList.at(i)->getPeerAddress().toString() << ".\n";
313  if(signalSender == m_cmdServer)
314  {
315  m_clientList.at(i)->setCmdSocket(newSocket);
316  qDebug() << "Added new Command-Socket to Client with ID " << m_clientList.at(i)->getClientID() << ".\n";
317  }
318  if(signalSender == m_dataServer)
319  {
320  m_clientList.at(i)->setDataSocket(newSocket);
321  qDebug() << "Added new Data-Socket to Client with ID " << m_clientList.at(i)->getClientID() << ".\n";
322  }
323  emit (newClient(m_clientList.at(i)->getClientID()));
324  return;
325  }
326  }

329  // Do not give a parent to newClient, so it can be moved to a thread

332 
333  if(signalSender == m_cmdServer)
